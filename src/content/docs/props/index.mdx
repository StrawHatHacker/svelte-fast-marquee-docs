## style
<PropDetail description="The inline style for the container div." type="string" defaultProp='""' />

<PropDetail description="The name of the css class to style the container div." type="string" defaultProp='""' />
## play
<PropDetail description="Whether to play or pause the marquee." type="boolean" defaultProp="true" />
## speed
<PropDetail description="The speed of the marquee in pixels per second." type="number" defaultProp="100" />
## gap
<PropDetail description='The distance between elements: "10px", "5%", "2rem"' type="string" defaultProp="0px" />
## pauseOnHover
<PropDetail description="Whether to pause the marquee when hovered." type="boolean" defaultProp="false" />
## pauseOnClick
<PropDetail description="Whether to pause the marquee when clicked." type="boolean" defaultProp="false" />
## direction
<PropDetail description="The direction the marquee is sliding." type='"left" or "right"' defaultProp='"left"' />
## gradient
<PropDetail description="Whether to show a gradient on right and left." type="boolean" defaultProp="false" />
## --gradientWidth
<PropDetail description='The width of the gradient on either side: "10px", "25%", "2vw"' type="string" defaultProp="10%" />
## --gradientColor
<PropDetail description='The color of the gradient: "red", "rgba(245, 245, 245, 0.5"), "hsl(0, 100%, 50%)"' type="string" defaultProp="white" />
